349  template < typename ComponentType >
350  ComponentType const* Entity::GetComponent() const
351  {
352  auto componentIterator = m_Components.find( typeid( ComponentType ) );
353  if ( componentIterator != m_Components.end() )
354  {
355  return static_cast< ComponentType* >( componentIterator->second );
356  }
357 
358  // if exact type not found, fall back to searching for a derived type
359  for ( auto const& [ type, component ] : m_Components )
360  {
361  ComponentType const* found = dynamic_cast< ComponentType const* >( component );
362  if ( found != nullptr )
363  {
364  return found;
365  }
366  }
367 
368  // if no derived component found, return nullptr
369  return nullptr;
370  }
