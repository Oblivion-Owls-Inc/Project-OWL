97 template < typename TileType >
98 glm::ivec2 Tilemap<TileType>::WorldPosToTileCoord( glm::vec2 pos ) const
99 {
100 const_cast< Tilemap< TileType >* >( this )->updateMat();
101
102 // convert to tilemap coordinate
103 glm::vec4 pos4(pos, 0, 1);
104 glm::ivec2 coord = glm::ivec2( m_InvMat * pos4 );
105
106 if (
107 coord.x < 0 || coord.y < 0 ||
108 coord.x >= m_Dimensions.x || coord.y >= m_Dimensions.y
109 )
110 {
111 return { -1, -1 };
112 }
113
114 return coord;
115 }

235 template < typename TileType >
236 void Tilemap<TileType>::updateMat()
237 {
238 if ( m_Transform == nullptr )
239 {
240 return;
241 }
242
243 // combine tilemap scale with parent transform position (will rotation ever be needed?)
244 m_Mat = glm::scale( glm::mat4( 1 ), { m_TileScale.x, -m_TileScale.y, 1 } );
245 glm::vec4 pos = { m_Transform->GetTranslation(), 0, 1 };
246 m_Mat[ 3 ] = pos;
247
248 m_InvMat = glm::inverse( m_Mat );
249 }

259 template < typename TileType >
260 void Tilemap<TileType>::readTilemap( nlohmann::ordered_json const& data )
261 {
262 if ( std::is_same_v< TileType , int > == false )
263 { // only serialize int tilemaps
264 return;
265 }
266
267 m_Dimensions.y = (int)data.size() / m_Dimensions.x;
268 int extraTiles = data.size() % m_Dimensions.x;
269 m_Tiles.resize( data.size() + ( extraTiles == 0 ? 0 : m_Dimensions.x - extraTiles ) );
270
271 m_Dimensions.y = (int)data.size() / m_Dimensions.x;
272 for ( int i = 0; i < data.size(); ++i )
273 {
274 Stream::Read( m_Tiles[ i ], data[ i ] );
275 }
276
277 callOnTilemapChangedCallbacks();
278 }

317 template < typename TileType >
318 nlohmann::ordered_json Tilemap<TileType>::Write() const
319 {
320 nlohmann::ordered_json data;
321
322 data[ "Dimensions" ] = Stream::Write( m_Dimensions );
323 data[ "TileScale" ] = Stream::Write( m_TileScale );
324
325 if ( std::is_same_v< TileType, int > == false )
326 {
327 return data;
328 }
329
330 nlohmann::ordered_json& tileData = data[ "TileData" ];
331 for ( TileType const& tile : m_Tiles )
332 {
333 tileData.push_back( Stream::Write( tile ) );
334 }
335
