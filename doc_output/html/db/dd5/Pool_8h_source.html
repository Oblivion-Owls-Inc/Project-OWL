252template<typename Value>
253bool Pool<Value>::Inspect()
254{
255 bool valueChanged = false;
256 float currentValue = static_cast<float>(m_CurrentValue);
257
258 // Display & edit the current value in the inspector.
259 if (ImGui::DragFloat("Current Value", &currentValue))
260 {
261 m_CurrentValue = static_cast<Value>(currentValue);
262 valueChanged = true;
263 }
264
265 // Display & edit the maximum value in the inspector.
266 float maximumValue = static_cast<float>(m_MaximumValue);
267 if (ImGui::DragFloat("Maximum Value", &maximumValue))
268 {
269 m_MaximumValue = static_cast<Value>(maximumValue);
270 valueChanged = true;
271 }
272
273 // Allows the reset function to be utilised in the inspector.
274 if (ImGui::Button("Reset"))
275 {
276 Reset();
277 valueChanged = true;
278 }
279
280 return valueChanged;
281}
