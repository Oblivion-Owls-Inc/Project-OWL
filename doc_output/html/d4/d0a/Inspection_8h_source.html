97 template< typename DataType >
98 bool Inspection::InspectArray(
99 char const* label,
100 std::vector< DataType >* array,
101 std::function< bool ( DataType* ) > elementInspector,
102 std::function< void ( DataType* ) > elementInitializer,
103 std::function< void ( DataType* ) > elementExiter
104 )
105 {
106
107 bool modified = false;
108
109 // get the ID of this ListBox to prevent dragging and dropping across different arrays
110 std::string fullId = label;
111 fullId += std::to_string( ImGui::GetItemID() );
112
113 ImGui::BeginTable( label, 3, ImGuiTableFlags_Borders );
114
115 ImGui::TableSetupColumn( "index", ImGuiTableColumnFlags_WidthFixed, 45.0f );
116 ImGui::TableSetupColumn( label );
117 ImGui::TableSetupColumn( "remove", ImGuiTableColumnFlags_WidthFixed, 30.0f );
118 ImGui::TableHeadersRow();
119
120 for ( int i = 0; i < array->size(); ++i )
121 {
122 ImGui::PushID( i );
123
124 ImGui::TableNextRow();
125 ImGui::TableNextColumn();
126
127 // allow dragging and dropping of the indices to reorder the array
128 ImGui::Text( "%3i", i );
129 if ( ImGui::BeginDragDropSource( ImGuiDragDropFlags_SourceAllowNullID ) )
130 {
131 ImGui::SetDragDropPayload( fullId.c_str(), &i, sizeof(int));
132
133 ImGui::Text( "%3i", i );
134
135 ImGui::EndDragDropSource();
136 }
137 if ( ImGui::BeginDragDropTarget() )
138 {
139 if ( ImGuiPayload const* payload = ImGui::AcceptDragDropPayload( fullId.c_str() ) )
140 {
141 if ( payload->DataSize != sizeof( int ) )
142 {
143 Debug() << "Warning: ImGui drag/drop payload does not match expected size" << std::endl;
144 }
145 else
146 {
147 int fromIndex = *(const int*)payload->Data;
148
149 DataType movedValue = (*array)[ fromIndex ];
150 array->erase( array->begin() + fromIndex );
151 array->insert( array->begin() + i, movedValue );
152
153 modified = true;
154 }
155 }
156 ImGui::EndDragDropTarget();
157 }
158
159 // inspect the element
160 ImGui::TableNextColumn();
161 if ( elementInspector( &(*array)[ i ] ) )
162 {
163 modified = true;
164 }
165
166 // button to delete the element
167 ImGui::TableNextColumn();
168 if ( ImGui::Button( "X", ImVec2( 23, 23 ) ) )
169 {
170 elementExiter( &(*array)[ i ] );
171 array->erase( array->begin() + i );
172 modified = true;
173 }
174
175 ImGui::PopID();
176 }
177
178 ImGui::TableNextRow();
179 ImGui::TableNextColumn();
180
181 // button to add element
182 if ( ImGui::Button( "+", ImVec2( 40, 23 ) ) )
183 {
184 array->push_back( DataType() );
185 elementInitializer( &array->back() );
186 modified = true;
187 }
188
189 ImGui::EndTable();
190
191 return modified;
192 }
