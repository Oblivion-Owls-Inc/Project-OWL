161 template < typename EventType >
162 void EventSystem::RemoveEventListener( EventListener< EventType >* eventListener )
163 {
164 // find the appropriate listener array
165 auto mapIterator = m_EventListeners.find( typeid( EventType ) );
166 if ( mapIterator == m_EventListeners.end() )
167 {
168 Debug() << "WARNING: there are no EventListeners of type \"" << typeid( EventType ).name() << "\" to remove" << std::endl;
169 return;
170 }
171 std::vector< EventListenerBase* >& listeners = mapIterator->second;
172
173 // find the appropriate listener in the array
174 auto vecIterator = std::find(
175 listeners.begin(),
176 listeners.end(),
177 static_cast< EventListener< EventType >* >( eventListener )
178 );
179 if ( vecIterator == listeners.end() )
180 {
181 Debug() << "WARNING: EventListener of type \"" << typeid( EventType ).name() << "\" could not be found to remove" << std::endl;
182 return;
183 }
184
185 // remove the listener
186 listeners.erase( vecIterator );
187
188 if ( listeners.empty() )
189 {
190 m_EventListeners.erase( mapIterator );
191 }
192 }
