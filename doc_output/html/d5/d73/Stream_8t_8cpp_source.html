388 template< typename ValueType >
389 void Stream::Read(ValueType& value, nlohmann::ordered_json const& json, IsISerializable<true>)
390 {
391 if (json.is_object() == false)
392 {
393 std::cerr << "JSON Error: unexpected type " << json.type_name() << " encountered while trying to read json Object of " <<
394 typeid(ValueType).name() << " at " << GetDebugLocation() << std::endl;
395 return;
396 }
397
398 ReadMethodMap< ISerializable > const& readMethods = value.GetReadMethods();
399 for (auto& [name, data] : json.items())
400 {
401 auto it = readMethods.find(name);
402 if (it == readMethods.end())
403 {
404 std::cerr << "JSON Warning: unrecognized token " << name << " encountered while trying to read " <<
405 typeid(ValueType).name() << " at " << GetDebugLocation() << std::endl;
406 continue;
407 }
408
409 s_DebugLocationStack.push_back( name + "." );
410
411 ReadMethod< ISerializable > const& readMethod = it->second;
412 (value.*readMethod)(data);
413
414 s_DebugLocationStack.pop_back();
415 }
416
417 value.AfterLoad();
418 }
