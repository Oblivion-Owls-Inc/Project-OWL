219 template<class AssetType>
220 void AssetLibrarySystem<AssetType>::ListAssets()
221 {
222 // list assets
223
224 for (auto ComponentIt = m_Assets.begin(); ComponentIt != m_Assets.end(); /* no increment here */)
225 {
226 bool erase = false;
227
228 if (!ImGui::TreeNode(ComponentIt->first.c_str()))
229 {
230 if (ImGui::BeginPopupContextItem(ComponentIt->first.c_str()))
231 {
232 if (ImGui::MenuItem("Copy"))
233 {
234 Stream::CopyToClipboard(ComponentIt->second);
235 }
236 if (ImGui::MenuItem("Paste"))
237 {
238 Stream::PasteFromClipboard(ComponentIt->second);
239 }
240 if (ImGui::MenuItem("Delete"))
241 {
242 Debug() << "Deleting " << ComponentIt->first << std::endl;
243 delete ComponentIt->second; // Delete the asset
244 erase = true;
245 // Don't continue here since we're erasing this element
246 }
247 ImGui::EndPopup();
248 }
249
250 if (!erase)
251 {
252 ++ComponentIt; // Only increment if we're not erasing this element
253 }
254 }
255 else
256 {
257 ComponentIt->second->Inspect();
258 ImGui::TreePop();
259 ++ComponentIt; // Increment here since no deletion is happening
260 }
261
262 if (erase)
263 {
264 ComponentIt = m_Assets.erase(ComponentIt);
265 }
266 }
267
268 }
